Advent of Code 2021, Day 1

import java.io.*; import java.util.*; public class AdventOne { public static void main(String[] args) throws FileNotFoundException { ArrayList nums = readFile("advent1.txt"); int count1 = 0; for(int i = 0; i < nums.size() - 1; i++) { if(nums.get(i) < nums.get(i+1)) { count1++; } } System.out.println("Part One: " + count1); //1559 //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ int count2 = 0; for(int i = 0; i < nums.size() - 3; i++) { if(nums.get(i) + nums.get(i+1) + nums.get(i+2) < nums.get(i+1) + nums.get(i+2) + nums.get(i+3)) { count2++; } } System.out.println("Part Two: " + count2); //1600 } public static ArrayList readFile(String fileName) throws FileNotFoundException { File inFile = new File(fileName); Scanner scanny = new Scanner(inFile); ArrayList nums = new ArrayList(); while(scanny.hasNextLine()) { nums.add(scanny.nextInt()); } scanny.close(); return nums; } }
