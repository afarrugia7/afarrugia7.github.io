Advent of Code 2021, Day 2


import java.io.*;
import java.util.*;

public class AdventTwo
{
   public static void main(String[] args) throws FileNotFoundException
   {
      //PART ONE
      ArrayList list = readFile("advent2.txt");
      int horizontal = 0;
      int depth = 0;
      
      for(int i = 0; i < list.size(); i++)
      {
         String[] arr = list.get(i);
         int num = Integer.parseInt(arr[1]);
         if(arr[0].equals("forward"))
         {
            horizontal += num;
         }
         else if(arr[0].equals("up"))
         {
            depth -= num;
         }
         else
         {
            depth += num;
         }
      }
      
      System.out.println("Part One: " + depth * horizontal); //2150351
      
      //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
      
      //PART TWO
      horizontal = 0;
      depth = 0;
      int aim = 0;
      for(int i = 0; i < list.size(); i++)
      {
         String[] arr = list.get(i);
         int num = Integer.parseInt(arr[1]);
         if(arr[0].equals("forward"))
         {
            horizontal += num;
            depth += (aim * num);
         }
         else if(arr[0].equals("up"))
         {
           aim -= num;
         }
         else
         {
            aim += num;
         }
      }
      System.out.println("Part Two: " + horizontal * depth); //1842742223
   }
   
   public static ArrayList readFile(String fileName) throws FileNotFoundException
   {
      File inFile = new File(fileName);
      Scanner scanny = new Scanner(inFile);
      ArrayList nums = new ArrayList();
      
      while(scanny.hasNextLine())
      {
         String[] arr = scanny.nextLine().split(" ");
         nums.add(arr);
      }
      scanny.close();
      return nums;
   }

}
